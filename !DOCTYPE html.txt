        const typeBadge = isIn 
          ? `<span class="bg-emerald-100 text-emerald-800 text-sm font-bold px-2.5 py-1 rounded-full border border-emerald-300">📈 입금</span>`
          : `<span class="bg-rose-100 text-rose-800 text-sm font-bold px-2.5 py-1 rounded-full border border-rose-300">📉 출금</span>`;
        
        const amtText = isIn 
          ? `<span class="text-emerald-700 font-bold">+${formatCurrency(item.amount)}</span>`
          : `<span class="text-rose-600 font-bold">-${formatCurrency(item.amount)}</span>`;

        listHtml += `
          <div class="bg-gray-50 border-2 border-gray-100 hover:border-[#FFB7B2]/40 rounded-2xl p-4 flex flex-col sm:flex-row justify-between items-start sm:items-center gap-3 transition-all shadow-sm">
            <div class="flex flex-col gap-1">
              <div class="flex items-center gap-2">
                <span class="text-gray-500 text-sm font-bold"><i class="fa-regular fa-calendar"></i> ${item.date}</span>
                ${typeBadge}
              </div>
              <div class="text-lg font-bold text-gray-800 mt-1">${escapeHtml(item.content)}</div>
            </div>
            
            <div class="flex items-center justify-between sm:justify-end w-full sm:w-auto gap-4">
              <div class="text-xl font-bold">${amtText}</div>
              <div class="flex gap-1">
                <button onclick="startEdit('${item.id}', '${item.date}', '${item.type}', '${item.content}', ${item.amount}, event)" 
                        class="effect-btn p-1.5 px-3 bg-amber-100 hover:bg-amber-200 text-amber-800 rounded-xl text-xs font-bold transition-all">
                  <i class="fa-solid fa-pen"></i> 수정
                </button>
                <button onclick="deleteItem('${item.id}', event)" 
                        class="effect-btn p-1.5 px-3 bg-rose-100 hover:bg-rose-200 text-rose-700 rounded-xl text-xs font-bold transition-all">
                  <i class="fa-solid fa-trash"></i> 삭제
                </button>
              </div>
            </div>
          </div>
        `;
      });

      container.innerHTML = listHtml;

      // 상단 잔액 카드 적용
      const balance = totalIn - totalOut;
      document.getElementById('totalIn').textContent = formatCurrency(totalIn);
      document.getElementById('totalOut').textContent = formatCurrency(totalOut);
      document.getElementById('totalBalance').textContent = formatCurrency(balance);

      // 목표금액 대비 소비 진단 상태 계산
      analyzeStatus(balance);
    }

    // 목표금액 분석 안내 메시지 도출
    function analyzeStatus(balance) {
      const msgElem = document.getElementById('statusMessage');
      if (balance >= monthlyGoalAmount) {
        msgElem.innerHTML = `🎉 대단해요! 목표 금액인 <b class="text-[#E58B88]">${formatCurrency(monthlyGoalAmount)}</b>을 돌파했습니다! 현재 보너스 <b>+${formatCurrency(balance - monthlyGoalAmount)}</b> 달성 중!`;
        msgElem.className = "mt-4 text-center bg-emerald-50 py-2 px-3 rounded-xl border border-emerald-300 text-sm text-emerald-800";
      } else {
        const diff = monthlyGoalAmount - balance;
        msgElem.innerHTML = `🐷 이번 달 목표 달성까지 <b class="text-[#E58B88]">${formatCurrency(diff)}</b> 남았습니다. 남은 기간 더 아껴볼까요? 파이팅!`;
        msgElem.className = "mt-4 text-center bg-amber-50 py-2 px-3 rounded-xl border border-amber-300 text-sm text-amber-800";
      }
    }

    // 입금/출금 타입 수동 선택 세팅
    function setType(type) {
      document.getElementById('txType').value = type;
      const btnIn = document.getElementById('btnTypeIn');
      const btnOut = document.getElementById('btnTypeOut');

      if (type === '입금') {
        btnIn.className = "py-2.5 rounded-2xl font-bold text-lg border-2 transition-all flex items-center justify-center gap-2 bg-emerald-50 border-emerald-300 text-emerald-700";
        btnOut.className = "py-2.5 rounded-2xl font-bold text-lg border-2 transition-all flex items-center justify-center gap-2 bg-white border-gray-200 text-gray-500 hover:bg-rose-50";
      } else {
        btnIn.className = "py-2.5 rounded-2xl font-bold text-lg border-2 transition-all flex items-center justify-center gap-2 bg-white border-gray-200 text-gray-500 hover:bg-emerald-50";
        btnOut.className = "py-2.5 rounded-2xl font-bold text-lg border-2 transition-all flex items-center justify-center gap-2 bg-rose-50 border-rose-300 text-rose-700";
      }
    }

    // 폼 등록 및 수정 트리거
    function handleFormSubmit(e) {
      e.preventDefault();
      
      const id = document.getElementById('editId').value;
      const date = document.getElementById('txDate').value;
      const type = document.getElementById('txType').value;
      const content = document.getElementById('txContent').value.trim();
      const amount = parseInt(document.getElementById('txAmount').value, 10);

      if (!date || !content || isNaN(amount) || amount <= 0) {
        showToast("⚠️ 정확한 양식값을 모두 채워주세요!", "error");
        return;
      }

      const itemPayload = { date, type, content, amount };

      if (id) {
        // 수정 모드
        showLoading("구글 시트의 내용을 고치는 중...");
        google.script.run
          .withSuccessHandler(function(updatedData) {
            transactionsData = updatedData;
            renderUI();
            resetForm();
            hideLoading();
            showToast("✏️ 내역이 정상적으로 수정되었습니다!");
          })
          .withFailureHandler(function() {
            hideLoading();
            showToast("🚨 수정 처리에 실패했습니다.", "error");
          })
          .updateTransaction(id, itemPayload);
      } else {
        // 추가 모드
        showLoading("구글 시트에 내역을 기록하는 중...");
        google.script.run
          .withSuccessHandler(function(updatedData) {
            transactionsData = updatedData;
            renderUI();
            resetForm();
            hideLoading();
            showToast("👛 저금통에 소중한 내역이 등록되었습니다!");
          })
          .withFailureHandler(function() {
            hideLoading();
            showToast("🚨 추가 처리에 실패했습니다.", "error");
          })
          .addTransaction(itemPayload);
      }
    }

    // 수정 모드 활성화 진입
    function startEdit(id, date, type, content, amount, e) {
      // 스크롤 이동
      document.getElementById('transactionForm').scrollIntoView({ behavior: 'smooth' });
      
      document.getElementById('formTitle').textContent = "기존 용돈 기록 수정하기";
      document.getElementById('editId').value = id;
      document.getElementById('txDate').value = date;
      document.getElementById('txContent').value = content;
      document.getElementById('txAmount').value = amount;
      setType(type);

      // 취소 버튼 노출 및 등록 버튼명 변환
      document.getElementById('submitBtn').innerHTML = "🛠️ 안전하게 수정하기";
      document.getElementById('submitBtn').className = "effect-btn flex-1 bg-amber-500 hover:bg-amber-600 text-white font-bold py-3 px-6 rounded-2xl text-xl shadow-md transition-all";
      document.getElementById('cancelEditBtn').classList.remove('hidden');
    }

    // 수정모드에서 돌아오기 (폼 비우기)
    function resetForm() {
      document.getElementById('formTitle').textContent = "새로운 용돈 거래 기록하기";
      document.getElementById('editId').value = "";
      document.getElementById('txDate').value = new Date().toISOString().split('T')[0];
      document.getElementById('txContent').value = "";
      document.getElementById('txAmount').value = "";
      setType('입금');

      document.getElementById('submitBtn').innerHTML = "💰 저금통에 쏙 넣기";
      document.getElementById('submitBtn').className = "effect-btn flex-1 bg-[#FFB7B2] hover:bg-[#FF9E99] text-white font-bold py-3 px-6 rounded-2xl text-xl shadow-md transition-all";
      document.getElementById('cancelEditBtn').classList.add('hidden');
    }

    // 데이터 삭제 처리
    function deleteItem(id, e) {
      if (!confirm("정말 이 내역을 삭제할까요?")) {
        return;
      }
      
      showLoading("구글 시트에서 안전하게 지우는 중...");
      google.script.run
        .withSuccessHandler(function(updatedData) {
          transactionsData = updatedData;
          renderUI();
          hideLoading();
          showToast("🗑️ 선택하신 기록이 완전히 지워졌습니다!");
        })
        .withFailureHandler(function() {
          hideLoading();
          showToast("🚨 삭제 처리에 실패했습니다.", "error");
        })
        .deleteTransaction(id);
    }

    // 목표 금액 저장 스크립트
    function saveGoal(e) {
      const gInput = document.getElementById('goalInput').value;
      const parsed = parseInt(gInput, 10);
      if (isNaN(parsed) || parsed < 0) {
        showToast("⚠️ 정확한 목표 금액 숫자를 입력하세요!", "error");
        return;
      }

      showLoading("새로운 한달 목표 금액을 설정하고 있어요...");
      google.script.run
        .withSuccessHandler(function() {
          monthlyGoalAmount = parsed;
          document.getElementById('goalDisplay').textContent = formatCurrency(parsed);
          renderUI();
          hideLoading();
          showToast("🎯 이번 달 목표 금액을 새롭게 다짐했습니다!");
        })
        .withFailureHandler(function() {
          hideLoading();
          showToast("🚨 목표금액 설정 저장을 실패했습니다.", "error");
        })
        .setMonthlyGoal(parsed);
    }

    // HTML 태그 탈옥 방지 안전 변환용 함수
    function escapeHtml(text) {
      return text
        .replace(/&/g, "&amp;")
        .replace(/</g, "&lt;")
        .replace(/>/g, "&gt;")
        .replace(/"/g, "&quot;")
        .replace(/'/g, "&#039;");
    }

    // 로딩 토글 유틸리티
    function showLoading(msg) {
      document.getElementById('loadingText').textContent = msg || "돼지 저금통 돌리는 중...";
      document.getElementById('loadingModal').classList.remove('hidden');
    }
    function hideLoading() {
      document.getElementById('loadingModal').classList.add('hidden');
    }

    // 토스트 알럿 노출 유틸리티
    function showToast(text, type = "success") {
      const toast = document.getElementById('customAlert');
      const icon = document.getElementById('alertIcon');
      const label = document.getElementById('alertText');

      if (type === "success") {
        icon.textContent = "🐷";
        toast.className = "fixed bottom-6 left-1/2 -translate-x-1/2 bg-emerald-600 text-white px-6 py-3 rounded-full shadow-2xl flex items-center gap-2 transition-all duration-300 transform z-50";
      } else {
        icon.textContent = "🚨";
        toast.className = "fixed bottom-6 left-1/2 -translate-x-1/2 bg-rose-600 text-white px-6 py-3 rounded-full shadow-2xl flex items-center gap-2 transition-all duration-300 transform z-50";
      }

      label.textContent = text;
      toast.style.opacity = "1";
      toast.style.transform = "translate(-50%, 0)";

      setTimeout(() => {
        toast.style.opacity = "0";
        toast.style.transform = "translate(-50%, 100px)";
      }, 3000);
    }

    // 인터랙티브 이펙트 함수 세팅 (마우스 돈 효과 & 버튼 팡팡 피드백)
    function setupInteractions() {
      const moneySymbols = ['🪙', '💵', '💰', '🐷', '✨'];
      const rainContainer = document.getElementById('money-rain-container');

      // 마우스 움직일 때마다 '돈 그림' 잔상 생성
      let lastMoveTime = 0;
      document.addEventListener('mousemove', (e) => {
        const now = Date.now();
        // 성능 과부하 방지를 위한 스로틀링 (80ms 마다 1개)
        if (now - lastMoveTime < 80) return;
        lastMoveTime = now;

        const moneyEl = document.createElement('div');
        moneyEl.className = 'floating-money';
        moneyEl.textContent = moneySymbols[Math.floor(Math.random() * moneySymbols.length)];
        moneyEl.style.left = e.clientX + 'px';
        moneyEl.style.top = e.clientY + 'px';
        
        rainContainer.appendChild(moneyEl);

        // 1.2초 뒤 가비지 컬렉션 처리
        setTimeout(() => {
          moneyEl.remove();
        }, 1200);
      });

      // 버튼을 누르면 사방으로 팡팡 터지는 파티클 효과 구현
      document.addEventListener('click', (e) => {
        // 클래스명이나 상위에 effect-btn이 있는 대상에서만 실행
        const target = e.target.closest('.effect-btn') || e.target.closest('button');
        if (!target) return;

        const colors = ['#FFB7B2', '#FFE066', '#FF9E99', '#4ADE80', '#FCD116'];
        
        for (let i = 0; i < 12; i++) {
          const particle = document.createElement('div');
          particle.className = 'particle';
          
          // 랜덤한 색상 및 사이즈 배정
          const color = colors[Math.floor(Math.random() * colors.length)];
          const size = Math.floor(Math.random() * 8) + 6;
          particle.style.width = size + 'px';
          particle.style.height = size + 'px';
          particle.style.backgroundColor = color;
          
          // 현재 마우스 클릭 위치
          particle.style.left = e.clientX + 'px';
          particle.style.top = e.clientY + 'px';
          
          // 방사형으로 튕겨 나갈 랜덤 좌표 계산
          const angle = Math.random() * Math.PI * 2;
          const velocity = Math.random() * 80 + 40;
          const tx = Math.cos(angle) * velocity;
          const ty = Math.sin(angle) * velocity;
          
          particle.style.setProperty('--tx', tx + 'px');
          particle.style.setProperty('--ty', ty + 'px');
          
          document.body.appendChild(particle);
          
          setTimeout(() => {
            particle.remove();
          }, 600);
        }
      });
    }
